import asyncio

from langsmith import Client


async def main():
    client = Client()
    project = await client.aread_project(project_name="default")
    trace_id = "<trace-id>"
    response = await client.traces.list_runs(
        trace_id,
        project_id=str(project.id),
        filter='eq(run_type, "llm")',
        selects=["NAME", "STATUS"],
    )
    llm_runs = response.items


asyncio.run(main())
